CREATE TABLE requests (
	reqID MEDIUMINT PRIMARY KEY NOT NULL AUTO_INCREMENT,
	username VARCHAR(20),
	reqTitle VARCHAR(40),
	priority TINYINT,
	program VARCHAR(20),
	site VARCHAR(20),
	description VARCHAR(200),
	reqDate DATE,
	approvalDate DATE,
	completionDate DATE,
	status VARCHAR(13)
);
